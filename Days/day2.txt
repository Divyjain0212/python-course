print() syntax:;
print(*objects,sep='',end='\n',file=sys.stdout(),flush=False)

*objects â†’ One or more things you want to print.
e.g. print("Hello", "World")   # Hello World

sep â†’ Separator between objects (default is space ' ').
print("A", "B", "C", sep="-")   # A-B-C

end â†’ What to print at the end (default is newline '\n')
print("Hello", end=" ")  
print("World")            # Output: Hello World

file â†’ Where to print (default is sys.stdout â†’ console).
with open("output.txt", "w") as f:
    print("Hello File", file=f)

flush â†’ Whether to force flushing the output buffer (default False).
print("Hello", flush=True)
